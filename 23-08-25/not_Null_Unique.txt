   sfname varchar(100) not null,
   slname varchar(100) not null,
   sdept varchar(100) not null,
   sad_no int not null,
   semail varchar(100) not null,
   unique (sid_no,sad_no)
);

desc course;

alter table course add column slname varchar(100);
-- Adding a constrains
alter table course modify column slname varchar(100) unique;
-- We are dropping the unique constrain , we can only drop the unique constrain 
alter table course drop index slname;
-- We can't drop the not null constrain directly because it is stored directly in the table , if no values are there it would take it as null 
-- The below command won't exceute 
alter table course modify column slname varchar(100) null;
