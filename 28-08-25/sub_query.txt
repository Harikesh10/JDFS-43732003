use stu123;
create table student (

    student_id INT PRIMARY KEY AUTO_INCREMENT,
    award_id int,
    student_name VARCHAR(50),
    student_dob DATE,
    stduent_email VARCHAR(100) UNIQUE,
    student_dept VARCHAR(50),
    student_yos INT
);


create table student_awards (
    student_id int,
    award_id INT PRIMARY KEY ,
    award_name VARCHAR(100),
    award_year int,
    award_category VARCHAR(50),
    awarded_by VARCHAR(100)

);
